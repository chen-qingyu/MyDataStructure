58 // Use the KMP algorithm to find the position of the pattern.
59 static int kmp(const char* this_str, const char* patt_str, int n, int m)
60 {
61 if (n < m)
62 {
63 return -1;
64 }
65
66 if (m == 0) // "" is in any string at index 0.
67 {
68 return 0;
69 }
70
71 int* match = new int[m];
72 match[0] = -1;
73
74 for (int j = 1; j < m; j++)
75 {
76 int i = match[j - 1];
77 while ((i >= 0) && (patt_str[i + 1] != patt_str[j]))
78 {
79 i = match[i];
80 }
81 match[j] = (patt_str[i + 1] == patt_str[j]) ? i + 1 : -1;
82 }
83
84 int s = 0;
85 int p = 0;
86 while (s < n && p < m)
87 {
88 if (this_str[s] == patt_str[p])
89 {
90 s++;
91 p++;
92 }
93 else if (p > 0)
94 {
95 p = match[p - 1] + 1;
96 }
97 else
98 {
99 s++;
100 }
101 }
102
103 delete[] match;
104
105 return (p == m) ? (s - m) : -1;
106 }

480 double to_decimal() const
481 {
482 // check infinity or nan
483 static const char* pos_infs[12] = {"inf", "INF", "Inf", "+inf", "+INF", "+Inf", "infinity", "INFINITY", "Infinity", "+infinity", "+INFINITY", "+Infinity"};
484 static const char* neg_infs[6] = {"-inf", "-INF", "-Inf", "-infinity", "-INFINITY", "-Infinity"};
485 static const char* nans[9] = {"nan", "NaN", "NAN", "+nan", "+NaN", "+NAN", "-nan", "-NaN", "-NAN"};
486
487 for (int i = 0; i < 12; ++i)
488 {
489 if (*this == pos_infs[i])
490 {
491 return INFINITY;
492 }
493 }
494 for (int i = 0; i < 6; ++i)
495 {
496 if (*this == neg_infs[i])
497 {
498 return -INFINITY;
499 }
500 }
501 for (int i = 0; i < 9; ++i)
502 {
503 if (*this == nans[i])
504 {
505 return NAN;
506 }
507 }

509 // not infinity or nan
510
511 double sign = 1; // default '+'
512 double decimal_part = 0;
513 int decimal_cnt = 0;
514 double exp_sign = 1; // default '+'
515 int exp_part = 0;
516
517 // FSM
518 state st = S_BEGIN_BLANK;
519 for (int i = 0; i < list_.size_; ++i)
520 {
521 event ev = get_event(list_.data_[i], 10);
522 switch (st | ev)
523 {
524 case S_BEGIN_BLANK | E_BLANK:
525 st = S_BEGIN_BLANK;
526 break;
527
528 case S_BEGIN_BLANK | E_SIGN:
529 sign = (list_.data_[i] == '+') ? 1 : -1;
530 st = S_SIGN;
531 break;
532
533 case S_BEGIN_BLANK | E_DEC_POINT:
534 case S_SIGN | E_DEC_POINT:
535 st = S_DEC_POINT_NOT_LEFT;
536 break;
537
538 case S_BEGIN_BLANK | E_NUMBER:
539 case S_SIGN | E_NUMBER:
540 case S_INT_PART | E_NUMBER:
541 decimal_part = decimal_part * 10 + char_to_integer(list_.data_[i], 10);
542 st = S_INT_PART;
543 break;
544
545 case S_INT_PART | E_DEC_POINT:
546 st = S_DEC_POINT_HAS_LEFT;
547 break;
548
549 case S_DEC_POINT_NOT_LEFT | E_NUMBER:
550 case S_DEC_PART | E_NUMBER:
551 case S_DEC_POINT_HAS_LEFT | E_NUMBER:
552 decimal_part = decimal_part * 10 + char_to_integer(list_.data_[i], 10);
553 decimal_cnt++;
554 st = S_DEC_PART;
555 break;
556
557 case S_INT_PART | E_EXP:
558 case S_DEC_POINT_HAS_LEFT | E_EXP:
559 case S_DEC_PART | E_EXP:
560 st = S_EXP;
561 break;
562
563 case S_EXP | E_SIGN:
564 exp_sign = (list_.data_[i] == '+') ? 1 : -1;
565 st = S_EXP_SIGN;
566 break;
567
568 case S_EXP | E_NUMBER:
569 case S_EXP_SIGN | E_NUMBER:
570 case S_EXP_PART | E_NUMBER:
571 exp_part = exp_part * 10 + char_to_integer(list_.data_[i], 10);
572 st = S_EXP_PART;
573 break;
574
575 case S_INT_PART | E_BLANK:
576 case S_DEC_POINT_HAS_LEFT | E_BLANK:
577 case S_DEC_PART | E_BLANK:
578 case S_EXP_PART | E_BLANK:
579 case S_END_BLANK | E_BLANK:
580 st = S_END_BLANK;
581 break;
582
583 default:
584 st = S_OTHER;
585 i = list_.size_; // exit loop
586 break;
587 }
588 }
589 if (st != S_INT_PART && st != S_DEC_POINT_HAS_LEFT && st != S_DEC_PART && st != S_EXP_PART && st != S_END_BLANK)
590 {
591 throw std::runtime_error("ERROR: Invalid literal for to_decimal().");
592 }
593
594 return sign * ((decimal_part / std::pow(10, decimal_cnt)) * std::pow(10, exp_sign * exp_part));
595 }
