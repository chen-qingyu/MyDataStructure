291 // Insert node.
292 Node* insert(Node* pos, const T& element)
293 {
294 if (pos == nullptr)
295 {
296 pos = new Node(element);
297 size_++;
298 }
299 else
300 {
301 if (element < pos->data_)
302 {
303 pos->link_left(insert(pos->left_, element));
304 }
305 else if (pos->data_ < element)
306 {
307 pos->link_right(insert(pos->right_, element));
308 }
309 }
310
311 return pos;
312 }
313
314 // Remove node.
315 Node* remove(Node* pos, const T& element)
316 {
317 if (pos)
318 {
319 if (element < pos->data_)
320 {
321 pos->link_left(remove(pos->left_, element));
322 }
323 else if (pos->data_ < element)
324 {
325 pos->link_right(remove(pos->right_, element));
326 }
327 else // element == pos->data_
328 {
329 if (pos->left_ && pos->right_) // certainly not the min or max pos
330 {
331 Node* tmp = find_min(pos->right_);
332 pos->data_ = tmp->data_;
333 pos->link_right(remove(pos->right_, tmp->data_));
334 }
335 else // may be the min or max pos
336 {
337 // if it is the min or max pos, mark it
338 if (pos == min_)
339 {
340 min_ = end_;
341 }
342 if (pos == max_) // must `if`, not `else if` cuz pos may be root
343 {
344 max_ = end_;
345 }
346
347 Node* tmp = pos;
348 pos = pos->left_ ? pos->left_ : pos->right_;
349 delete tmp;
350 size_--;
351 }
352 }
353 }
354
355 return pos;

369 // Traverse the subtree rooted at the specified node.
370 template <typename F>
371 static void level_action(Node* node, const F& action)
372 {
373 // level order
374 if (node != nullptr)
375 {
376 Deque<Node*> queue;
377 queue.push_back(node);
378 while (!queue.is_empty())
379 {
380 node = queue.pop_front();
381 action(node->data_);
382 if (node->left_)
383 {
384 queue.push_back(node->left_);
385 }
386 if (node->right_)
387 {
388 queue.push_back(node->right_);
389 }
390 }
